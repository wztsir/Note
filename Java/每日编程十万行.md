int count(int low, int high) {
    if (low > high) return 1;

    if (res[low][high] != 0) return res[low][high];
    int out = 0;
    for (int i = low; i <= high; i++) {

        out += count(low, i - 1) * count(i + 1, high);

    }
    res[low][high] = out;
    return out;
}
